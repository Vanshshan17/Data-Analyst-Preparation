WITH temp_tt1 AS(
SELECT player_id,MIN(event_date) AS first_date FROM Activity GROUP BY player_id),

temp_tt2 AS(
SELECT *,DATE_ADD(first_date,INTERVAL 1 DAY) AS next_date FROM temp_tt1)

SELECT ROUND((SELECT COUNT(DISTINCT player_id) FROM Activity WHERE (player_id,event_date) IN (SELECT player_id,next_date FROM temp_tt2))
/(SELECT COUNT(DISTINCT player_id) FROM Activity),2) AS fraction;
